@override
Pages fetchPages(document, [String? searchColumn, String? searchWord]) {
  final Pages results = createPages(searchColumn, searchWord);

  final int lastPage = 16;
  for (int i = 1; i <= lastPage; i++) {
    final int page = i;
    results['pageMetas'].add({'page': page});
  }
  return results;
}
