@override
Future<void> init() async {
  try {
    // 1. Fetch 간격 및 타임아웃 설정
    await _remoteConfig.setConfigSettings(RemoteConfigSettings(
      fetchTimeout: const Duration(minutes: 1),
      // kDebugMode일 때는 즉시 반영되도록 0초, 릴리즈 모드일 때는 1시간 권장
      minimumFetchInterval:
          kDebugMode ? const Duration(seconds: 0) : const Duration(hours: 1),
    ));

    // 2. 기본값(Default) 설정
    await _remoteConfig.setDefaults(const {
      "min_app_version": "1.0.0",
      "show_notice_banner": false,
    });

    // 3. 서버 데이터 가져오고 활성화하기
    await _remoteConfig.fetchAndActivate();
  } catch (e) {
    debugPrint('Firebase Remote Config 초기화 실패: $e');
    // 필요에 따라 Exception을 던져서 Repository에서 처리하도록 할 수 있어.
  }
}
